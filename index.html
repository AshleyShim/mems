    // ========================================

    // Main Application Logic

    const app = {

      currentPhoto: null,


      init() {

        console.log('App initializing...');

        

        // Set today's date as default

        const today = new Date().toISOString().split('T')[0];

        const dateInput = document.getElementById('memDate');

        if (dateInput) {

          dateInput.value = today;

        }


        // Photo upload handler

        const photoInput = document.getElementById('photoInput');

        if (photoInput) {

          photoInput.addEventListener('change', this.handlePhotoUpload.bind(this));

        }

      },


      // Handle photo upload and preview

      handlePhotoUpload(e) {

        const file = e.target.files[0];

        if (file) {

          const reader = new FileReader();

          reader.onload = (event) => {

            this.currentPhoto = event.target.result;

            const preview = document.getElementById('photoPreview');

            preview.src = this.currentPhoto;

            preview.classList.remove('hidden');

          };

          reader.readAsDataURL(file);

        }

      },


      // Handle form submission

      handleFormSubmit(e) {

        if (e) e.preventDefault();

        console.log('Form submitted!');


        // Check if at least one field has content

        const hasPhoto = this.currentPhoto !== null;

        const hasDate = document.getElementById('memDate').value.trim() !== '';

        const hasPeople = document.getElementById('memPeople').value.trim() !== '';

        const hasQuote = document.getElementById('memQuote').value.trim() !== '';

        const hasFood = document.getElementById('memFood').value.trim() !== '';

        const hasNotes = document.getElementById('memNotes').value.trim() !== '';


        console.log('Has content:', { hasPhoto, hasDate, hasPeople, hasQuote, hasFood, hasNotes });


        if (!hasPhoto && !hasDate && !hasPeople && !hasQuote && !hasFood && !hasNotes) {

          this.showConfirmation('Please add at least one thing to remember! 📝', 'Oops!');

          return;

        }


        const mem = {

          id: Date.now(),

          photo: this.currentPhoto,

          date: document.getElementById('memDate').value,

          people: document.getElementById('memPeople').value,

          quote: document.getElementById('memQuote').value,

          food: document.getElementById('memFood').value,

          notes: document.getElementById('memNotes').value,

          createdAt: new Date().toISOString()

        };


        console.log('Saving memory:', mem);


        // Save to localStorage

        this.saveMem(mem);


        // Reset form

        document.getElementById('memForm').reset();

        document.getElementById('photoPreview').classList.add('hidden');

        this.currentPhoto = null;


        // Show success confirmation

        this.showConfirmation('Your moment has been safely tucked away', 'Memory Saved!');

      },


      // Save memory (persistent)

      saveMem(mem) {

        const mems = this.getMems();

        mems.push(mem);

        localStorage.setItem('memsData', JSON.stringify(mems));

      },


      // Get all memories (persistent)

      getMems() {

        try {

          return JSON.parse(localStorage.getItem('memsData')) || [];

        } catch (e) {

          return [];

        }

      },


      // Display random memory

      showRandomMem() {

        const mems = this.getMems();

        

        if (mems.length === 0) {

          this.showEmptyState();

          return;

        }


        // Get a random memory that's different from the current one

        let randomMem;

        if (mems.length === 1) {

          randomMem = mems[0];

        } else {

          do {

            randomMem = mems[Math.floor(Math.random() * mems.length)];

          } while (randomMem.id === this.currentMemId && mems.length > 1);

        }

        

        this.currentMemId = randomMem.id;

        this.displayMem(randomMem);

      },


      // Display a memory

      displayMem(mem) {

        const content = document.getElementById('memoryContent');

        

        let html = '<div class="memory-card">';


        if (mem.photo) {

          html += `<img src="${mem.photo}" alt="Memory" class="memory-photo">`;

        }


        if (mem.date) {

          const formattedDate = new Date(mem.date).toLocaleDateString('en-US', {

            weekday: 'long',

            year: 'numeric',

            month: 'long',

            day: 'numeric'

          });

          html += `

            <div class="memory-detail">

              <div class="memory-label">Date</div>

              <div class="memory-value">${formattedDate}</div>

            </div>

          `;

        }


        if (mem.people) {

          html += `

            <div class="memory-detail">

              <div class="memory-label">With</div>

              <div class="memory-value">${mem.people}</div>

            </div>

          `;

        }


        if (mem.quote) {

          html += `

            <div class="memory-detail">

              <div class="memory-label">Quote</div>

              <div class="memory-value">"${mem.quote}"</div>

            </div>

          `;

        }


        if (mem.food) {

          html += `

            <div class="memory-detail">

              <div class="memory-label">Favorite Bite</div>

              <div class="memory-value">${mem.food}</div>

            </div>

          `;

        }


        if (mem.notes) {

          html += `

            <div class="memory-detail">

              <div class="memory-label">Notes</div>

              <div class="memory-value">${mem.notes}</div>

            </div>

          `;

        }


        html += `

          <button class="btn share-btn" onclick="app.shareMem()">

            <span class="btn-icon">✨</span>

            Share this memory

          </button>

          <button class="btn share-btn" onclick="app.showRandomMem()" style="margin-top: 12px;">

            <span class="btn-icon">🎲</span>

            Show another memory

          </button>

        </div>`;


        content.innerHTML = html;

      },


      // Empty state

      showEmptyState() {

        const content = document.getElementById('memoryContent');

        content.innerHTML = `

          <div class="memory-card">

            <div class="empty-state">

              <div class="empty-icon">📭</div>

              <h2 class="empty-title">No memories yet</h2>

              <p class="empty-message">Start recording moments to fill your keepsake</p>

              <button class="btn" onclick="app.showRecord()" style="margin-top: 28px;">

                <span class="btn-icon">✏️</span>

                Record your first memory

              </button>

            </div>

          </div>

        `;

      },


      // Share memory

      shareMem() {

        this.showConfirmation('Take a screenshot to share your memory with friends!', 'Share Feature Coming Soon');

      },


      // Show confirmation overlay

      showConfirmation(message, title = 'Memory Saved!') {

        const overlay = document.getElementById('confirmationOverlay');

        const titleEl = overlay.querySelector('.confirmation-title');

        const messageEl = overlay.querySelector('.confirmation-message');

        

        titleEl.textContent = title;

        messageEl.textContent = message;

        

        overlay.classList.remove('hidden');

      },


      // Close confirmation

      closeConfirmation() {

        document.getElementById('confirmationOverlay').classList.add('hidden');

        this.showHome();

      },

      showHome() {

        document.getElementById('homeView').style.display = 'block';

        window.scrollTo(0, 0);

      showMems() {

        document.getElementById('recordView').classList.remove('active');
